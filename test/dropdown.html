Change dropdown valuechanges Back
Disable dropdown if selected 


private subscriptions: Subscription[] = [];

this.filterGroup = this.fb.group({
      
  client_id: [],
  monthselect: []
});

this.subscriptions.push(
  this.filterGroup.controls.client_id.valueChanges.subscribe(() => this.filter())
);
this.subscriptions.push(
  this.filterGroup.controls.monthselect.valueChanges.subscribe(() => this.filter())
);


ngOnDestroy() {
  this.subscriptions.forEach((sb) => sb.unsubscribe());
}

//get value
const filter = {};
const client_id = this.filterGroup.get('client_id').value;
if (client_id) {
  filter['client_id'] = client_id;
}

const monthselect = this.filterGroup.get('monthselect').value;
if (monthselect) {
  filter['monthselect'] = monthselect;
}
