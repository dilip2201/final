categoryList: any;
selectedAll: any;

fetchtasks(search = "") {
  this.workflowservice.getalltask(this.data.workflowid, this.data.taskid, search).subscribe((tasks: any) => {
    this.cd.markForCheck();
    if (tasks.data.length) {
      var result = tasks.data.map(function (el) {
        var o = Object.assign({}, el);
        o.isSelected = false;
        return o;
      })
      this.categoryList = result;
    }
  });
}


selectAll() {
  for (var i = 0; i < this.categoryList.length; i++) {
    this.categoryList[i].isSelected = this.selectedAll;
  }
}


checkIfAllSelected() {
  this.selectedAll = this.categoryList.every(function (item: any) {
    return item.isSelected == true;
  })
}
